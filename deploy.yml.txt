name: CI/CD Pipeline

on:
  push:
    branches: [ main ]
  pull_request:
    branches: [ main ]

jobs:
  build-and-test:
    runs-on: ubuntu-latest

    steps:
    - name: Checkout Source Code
      uses: actions/checkout@v4

    - name: Setup Node.js Runtime
      uses: actions/setup-node@v4
      with:
        node-version: '18'
        cache: 'npm'
        cache-dependency-path: './backend/package-lock.json'

    - name: Install Backend Dependencies
      run: |
        cd backend
        npm install

    - name: Build Docker Container (Backend Validation)
      run: |
        docker build -t restaurant-backend:test ./backend

    # Replace this block with your cloud-specific deployment actions (e.g., AWS, Render, Vercel)
    - name: Production Deployment Mock
      run: echo "Production environment build successful, ready for cloud release."
